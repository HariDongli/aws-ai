static Future<String> detectText(String accessKey, String secretKey, String region, File imagefile) async {

  try {
    List<int> imageBytes = imagefile.readAsBytesSync();
    String base64Image = base64Encode(imageBytes);
    String body = '{"Image":{"Bytes": "$base64Image"}}';
    String amzTarget = "RekognitionService.DetectText";

    String response  = await _rekognitionHttp(accessKey, secretKey, region, amzTarget, body);
    return response;
  }
  catch(e)
  {
    print(e);
    return "{}";
  }

  /*
  Output will be in this format

                "BoundingBox": {

                   {
                      "X": number,
                      "Y": number
                   }
                ]
             },
             "Id": number,
             "ParentId": number,
             "Type": "string"
          }
       ]
    }
  */

}
